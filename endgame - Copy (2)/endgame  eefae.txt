def generate_ppt():
    """Generate a PowerPoint presentation from voice input."""
    speak("Please say the topic for your PowerPoint presentation.")
    topic = command()
    
    if not topic:
        speak("No topic detected. Please try again.")
        return

    speak(f"Generating a presentation on {topic}. Please wait.")
    
    try:
        # Generate content using Gemini AI
        prompt = f"Create an educational PowerPoint outline on {topic}. Include a title slide and 4-5 key slides with summarized points."
        response = model.generate_content(prompt)
        slides_content = response.text.split("\n\n")  # Split text into slides

        # Create PowerPoint
        ppt = Presentation()

        # Title Slide
        slide_layout = ppt.slide_layouts[0]
        slide = ppt.slides.add_slide(slide_layout)
        title = slide.shapes.title
        subtitle = slide.placeholders[1]
        title.text = f"{topic}"
        subtitle.text = "Generated by Jarvis AI"

        # Add Content Slides
        for slide_text in slides_content:
            slide_layout = ppt.slide_layouts[1]  # Title & Content Layout
            slide = ppt.slides.add_slide(slide_layout)
            title, content = slide.shapes.title, slide.placeholders[1]

            slide_parts = slide_text.split(": ", 1)
            if len(slide_parts) == 2:
                title.text, content.text = slide_parts
            else:
                title.text, content.text = topic, slide_text

        # Save the presentation
        ppt_filename = f"{topic.replace(' ', '_')}.pptx"
        ppt.save(ppt_filename)

        speak(f"Presentation on {topic} has been created successfully. Opening the file now.")
        os.startfile(ppt_filename)

    except Exception as e:
        speak("I encountered an error while generating the presentation.")
        print("Error:", e)

def command():
    r = sr.Recognizer()
    with sr.Microphone() as source:
        print("Listening...")
        r.pause_threshold = 1  # Increased pause threshold for better sentence completion
        r.energy_threshold = 300  # Lower energy threshold to detect softer speech
        r.adjust_for_ambient_noise(source, duration=0.5)  # Adjust for ambient noise
        audio = r.listen(source)
    try:
        content = r.recognize_google(audio, language='en-in')
        print("You said:", content)
        return content.lower()
    except sr.UnknownValueError:
        print("Could not understand audio")
        return ""
    except Exception as e:
        print(f"Error recognizing speech: {e}")
        speak("Please try again")
        return ""

# Dictionary
def get_definition(word):
    synsets = wordnet.synsets(word)
    if synsets:
        definition = synsets[0].definition()
        return definition
    return "Sorry, I couldn't find the definition."

def matches_command(request, keywords):
    """Check if the request contains any of the keywords."""
    return any(keyword in request for keyword in keywords)

class MovableMainWindow(QtWidgets.QMainWindow):
    def __init__(self):
        super(MovableMainWindow, self).__init__()
        self.oldPos = None
    
    def mousePressEvent(self, event):
        self.oldPos = event.globalPos()

    def mouseMoveEvent(self, event):
        if self.oldPos:
            delta = QtCore.QPoint(event.globalPos() - self.oldPos)
            self.move(self.x() + delta.x(), self.y() + delta.y())
            self.oldPos = event.globalPos()

class FaceRecognitionThread(QThread):
    update_frame = QtCore.pyqtSignal(QImage)
    authentication_result = QtCore.pyqtSignal(bool)
    
    def __init__(self, reference_image_path):
        super().__init__()
        self.reference_image_path = reference_image_path
        self.running = True
        
    def run(self):
        # Load face cascade for simple face detection - no dlib required
        face_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_frontalface_default.xml')
        
        # Access webcam (use camera index 1 for DroidCam)
        cap = cv2.VideoCapture(0)  # Changed from 0 to 1 for DroidCam
        if not cap.isOpened():
            print("Cannot open camera")
            self.authentication_result.emit(False)
            return
            
        face_detected_frames = 0
        required_frames = 20  # Need to detect face in 20 frames to authenticate
        attempts = 0
        max_attempts = 100  # About 10 seconds of trying
        
        while self.running and attempts < max_attempts:
            ret, frame = cap.read()
            if not ret:
                print("Can't receive frame. Exiting...")
                break
                
            # Convert to grayscale for face detection
            gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
            
            # Detect faces
            faces = face_cascade.detectMultiScale(gray, 1.3, 5)
            
            # Add visual effects to frame
            # Add a border
            frame = cv2.copyMakeBorder(frame, 10, 10, 10, 10, cv2.BORDER_CONSTANT, value=[0, 120, 255])
            
            # Add scanning effect overlay
            overlay = frame.copy()
            # Draw horizontal scan line
            scan_position = (attempts % 50) * (frame.shape[0] // 50)
            cv2.line(overlay, (0, scan_position), (frame.shape[1], scan_position), (0, 255, 0), 2)
            frame = cv2.addWeighted(overlay, 0.7, frame, 0.3, 0)
            
            # Add system status text
            cv2.putText(frame, "JARVIS SECURITY SYSTEM", (frame.shape[1]//2 - 150, 30), 
                      cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 255), 2)
            
            if len(faces) > 0:
                # Face detected
                face_detected_frames += 1
                
                # Draw rectangle around face
                for (x, y, w, h) in faces:
                    # Draw fancy rectangle
                    cv2.rectangle(frame, (x, y), (x+w, y+h), (0, 255, 0), 2)
                    
                    # Draw corner lines for futuristic look
                    corner_length = 20
                    # Top left
                    cv2.line(frame, (x, y), (x + corner_length, y), (0, 255, 255), 2)
                    cv2.line(frame, (x, y), (x, y + corner_length), (0, 255, 255), 2)
                    # Top right
                    cv2.line(frame, (x+w, y), (x+w - corner_length, y), (0, 255, 255), 2)
                    cv2.line(frame, (x+w, y), (x+w, y + corner_length), (0, 255, 255), 2)
                    # Bottom left
                    cv2.line(frame, (x, y+h), (x + corner_length, y+h), (0, 255, 255), 2)
                    cv2.line(frame, (x, y+h), (x, y+h - corner_length), (0, 255, 255), 2)
                    # Bottom right
                    cv2.line(frame, (x+w, y+h), (x+w - corner_length, y+h), (0, 255, 255), 2)
                    cv2.line(frame, (x+w, y+h), (x+w, y+h - corner_length), (0, 255, 255), 2)
                    
                    progress = min(face_detected_frames / required_frames, 1.0)
                    bar_width = int(w * progress)
                    
                    # Draw progress bar
                    cv2.rectangle(frame, (x, y+h+5), (x+bar_width, y+h+15), (0, 255, 0), -1)
                    cv2.rectangle(frame, (x, y+h+5), (x+w, y+h+15), (255, 255, 255), 1)
                    
                    # Progress percentage
                    percentage = int(progress * 100)
                    cv2.putText(frame, f"Scanning: {percentage}%", (x, y+h+30), 
                              cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
                
                # Check if we have enough consistent detections
                if face_detected_frames >= required_frames:
                    # Authentication successful
                    # Create success overlay
                    success_overlay = frame.copy()
                    cv2.rectangle(success_overlay, (0, 0), (frame.shape[1], frame.shape[0]), (0, 255, 0), -1)
                    
                    # Blend with original frame
                    frame = cv2.addWeighted(success_overlay, 0.3, frame, 0.7, 0)
                    
                    # Add success text
                    cv2.putText(frame, "ACCESS GRANTED", (frame.shape[1]//2 - 150, frame.shape[0]//2), 
                              cv2.FONT_HERSHEY_SIMPLEX, 1.5, (255, 255, 255), 3)
                    cv2.putText(frame, "AUTHENTICATION SUCCESSFUL", (frame.shape[1]//2 - 200, frame.shape[0]//2 + 50), 
                              cv2.FONT_HERSHEY_SIMPLEX, 1, (255, 255, 255), 2)
                    
                    # Convert frame to QImage and emit
                    h, w, ch = frame.shape
                    qt_image = QImage(frame.data, w, h, ch * w, QImage.Format_RGB888)
                    self.update_frame.emit(qt_image.rgbSwapped())
                    
                    # Wait a moment to display the success message
                    time.sleep(1.5)
                    
                    # Authentication successful - speak confirmation
                    speak("Authentication successful. Welcome back, sir.")
                    
                    # Authentication successful
                    cap.release()
                    self.authentication_result.emit(True)
                    return
            else:
                # Reset counter if face is lost
                face_detected_frames = max(0, face_detected_frames - 1)
                # Warning text
                cv2.putText(frame, "No face detected", (50, 50), cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 0, 255), 2)
                cv2.putText(frame, "Please position your face in the camera view", (50, 80), 
                          cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 0, 255), 2)
            
            # Add scanning line effect
            
            # Convert frame to QImage and emit
            h, w, ch = frame.shape
            qt_image = QImage(frame.data, w, h, ch * w, QImage.Format_RGB888)
            self.update_frame.emit(qt_image.rgbSwapped())
            
            attempts += 1
            time.sleep(0.1)
            
        # If we get here without successful authentication
        cap.release()
        speak("Authentication failed. Access denied.")
        self.authentication_result.emit(False)
    
    def stop(self):
        self.running = False
        self.wait()

class FaceAuthenticationWindow(QtWidgets.QMainWindow):

        self.status_label = QtWidgets.QLabel("Initializing facial recognition system...")
        self.status_label.setStyleSheet("font-size: 18px; color: #00A9F4; margin-top: 10px;")
        self.status_label.setAlignment(QtCore.Qt.AlignCenter)
        main_layout.addWidget(self.status_label)
        
        # Start face recognition with voice prompt
        speak("Initializing security systems")
        QTimer.singleShot(1500, self.start_face_recognition)
    
    def start_face_recognition(self):
        speak("Beginning facial authentication")
        self.status_label.setText("Looking for your face...")
        # Create and start the face recognition thread
        self.face_thread = FaceRecognitionThread(self.reference_image_path)
        self.face_thread.update_frame.connect(self.update_frame)
        self.face_thread.authentication_result.connect(self.handle_authentication)
        self.face_thread.start()
    
    def update_frame(self, qt_image):
        pixmap = QPixmap.fromImage(qt_image)
        self.video_label.setPixmap(pixmap.scaled(self.video_label.width(), self.video_label.height(), QtCore.Qt.KeepAspectRatio))
    
    def handle_authentication(self, success):
        if success:
            self.status_label.setText("Authentication Successful! Launching JARVIS...")
            self.status_label.setStyleSheet("font-size: 18px; color: #00FF00; margin-top: 10px;")
            # Wait a moment to display the success message
            QTimer.singleShot(2000, self.launch_jarvis)
        else:
            self.status_label.setText("Authentication Failed! Access Denied.")
            self.status_label.setStyleSheet("font-size: 18px; color: #FF0000; margin-top: 10px;")
            # Wait a moment and then close the application
            QTimer.singleShot(3000, self.close)
    
    def launch_jarvis(self):
        # Close the authentication window
        self.close()
        # Launch the Jarvis application
        app = QtWidgets.QApplication.instance() or QtWidgets.QApplication(sys.argv)
        MainWindow = MovableMainWindow()
        ui = Ui_MainWindow()
        ui.setupUi(MainWindow)
        MainWindow.show()
        sys.exit(app.exec_())
    
    def closeEvent(self, event):
        # Stop the face recognition thread when closing the window
        if hasattr(self, 'face_thread'):
            self.face_thread.stop()
        event.accept()

def save_reference_image():
    """Capture and save a reference image of the user's face"""
    if os.path.exists("face_reference.jpg"):
        return  # Reference image already exists
        
    print("Capturing reference image of your face...")
    cap = cv2.VideoCapture(0)  # Changed from 0 to 1 for DroidCam
    if not cap.isOpened():
        print("Cannot open camera")
        return False
        
    # Wait a bit to let the camera adjust
    for _ in range(10):
        cap.read()
        time.sleep(0.1)
    
    ret, frame = cap.read()
    cap.release()
    
    if ret:
        # Save the reference image
        cv2.imwrite("face_reference.jpg", frame)
        print("Reference image saved as 'face_reference.jpg'")
        return True
    else:
        print("Failed to capture reference image")
        return False
